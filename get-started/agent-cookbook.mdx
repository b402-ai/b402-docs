## 1. Simple EOA Payment

Pay for an API call using a standard wallet with EIP-712 signatures. Requires a one-time token approval.

```ts
import { Wallet, Contract, JsonRpcProvider } from 'ethers';
import { B402 } from 'b402-sdk';

const RPC = 'https://bsc-dataseed1.binance.org';
const USD1 = '0x8d0d000ee44948fc98c9b98a4fa4921476f08b0d';
const RELAYER = '0x590162bf4b50F6576a459B75309eE21D92178A10';

async function eoaPayment() {
  const provider = new JsonRpcProvider(RPC);
  const wallet = new Wallet(process.env.PRIVATE_KEY!, provider);

  // One-time approval (skip if already approved)
  const token = new Contract(
    USD1,
    [
      'function allowance(address,address) view returns (uint256)',
      'function approve(address,uint256) returns (bool)',
    ],
    wallet
  );

  const allowance = await token.allowance(wallet.address, RELAYER);
  if (allowance === 0n) {
    const tx = await token.approve(
      RELAYER,
      '0xffffffffffffffffffffffffffffffffffffffffffffffffffffffffffffffff'
    );
    await tx.wait();
    console.log('Relayer approved');
  }

  // Send payment
  const b402 = new B402('mainnet');
  const result = await b402.pay(wallet, {
    amount: '0.01',
    token: 'USD1',
    recipient: '0x...seller_address',
  });

  console.log('TX:', result.txHash);
  // TX: 0x1234...abcd
}

eoaPayment();
```

**Expected output:**
```
Relayer approved
TX: 0x1234...abcd
```

## 2. Smart Wallet Payment

Gasless payment from a smart wallet. No approval step, no gas tokens needed.

```ts
import { Wallet } from 'ethers';

const FACILITATOR = 'https://facilitator.b402.ai';
const USDT = '0x55d398326f99059fF775485246999027B3197955';

async function smartWalletPayment() {
  const owner = new Wallet(process.env.PRIVATE_KEY!);

  // Get smart wallet address
  const walletRes = await fetch(
    `${FACILITATOR}/wallet/${owner.address}/master`
  );
  const { walletAddress } = await walletRes.json();
  console.log('Smart wallet:', walletAddress);

  // Verify: get unsigned UserOp
  const verifyRes = await fetch(`${FACILITATOR}/wallet/verify`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      walletAddress,
      transactions: [
        {
          to: '0x...seller_address',
          amount: '10000000000000000', // 0.01 USDT
          token: USDT,
        },
      ],
      paymentRequirements: { network: 'bsc' },
    }),
  });
  const { userOp, userOpHash } = await verifyRes.json();

  // Sign the UserOp hash
  const signature = await owner.signMessage(
    Buffer.from(userOpHash.slice(2), 'hex')
  );

  // Settle: submit signed UserOp
  const settleRes = await fetch(`${FACILITATOR}/wallet/settle`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      userOp: { ...userOp, signature },
      signature,
    }),
  });

  const result = await settleRes.json();
  console.log('TX:', result.txHash);
  // TX: 0xabcd...1234
}

smartWalletPayment();
```

**Expected output:**
```
Smart wallet: 0x...deterministic_address
TX: 0xabcd...1234
```

## 3. Express Middleware Setup

Accept b402 payments as a seller. The middleware handles 402 responses, verification, and settlement automatically.

```ts
import express from 'express';
import { b402 } from 'b402-express';

const app = express();

// Configure paid routes
app.use(
  b402(
    { payTo: '0x...your_wallet_address' },
    {
      'GET /api/premium': { price: '0.01', token: 'USD1' },
      'GET /api/data': { price: '0.005', token: 'USDT' },
      'POST /api/generate': { price: '0.05', token: 'USDC' },
    }
  )
);

// Paid endpoint - payment verified before handler runs
app.get('/api/premium', (req, res) => {
  console.log('Paid by:', req.b402.payer);
  console.log('TX:', req.b402.txHash);
  res.json({
    data: 'premium content',
    payer: req.b402.payer,
  });
});

// Free endpoint - no payment required
app.get('/api/health', (_req, res) => {
  res.json({ status: 'ok' });
});

app.listen(3000, () => {
  console.log('b402 server running on port 3000');
});
```

**When a buyer calls a paid endpoint without payment:**
```json
{
  "status": 402,
  "paymentRequirements": {
    "token": "0x8d0d000ee44948fc98c9b98a4fa4921476f08b0d",
    "amount": "10000000000000000",
    "payTo": "0x...your_wallet_address",
    "relayerContract": "0x590162bf4b50F6576a459B75309eE21D92178A10",
    "network": "bsc"
  }
}
```

**After valid payment:**
```json
{
  "data": "premium content",
  "payer": "0x...buyer_address"
}
```

## 4. Multi-Token Support

Handle payments in USDT, USDC, or USD1. The token address determines which stablecoin is used.

```ts
import { Wallet } from 'ethers';

const FACILITATOR = 'https://facilitator.b402.ai';

// All tokens use 18 decimals on BNB Chain
const TOKENS: Record<string, { address: string; decimals: number }> = {
  USDT: {
    address: '0x55d398326f99059fF775485246999027B3197955',
    decimals: 18,
  },
  USDC: {
    address: '0x8AC76a51cc950d9822D68b83fE1Ad97B32Cd580d',
    decimals: 18,
  },
  USD1: {
    address: '0x8d0d000ee44948fc98c9b98a4fa4921476f08b0d',
    decimals: 18,
  },
};

function toWei(amount: string, decimals: number): string {
  const [whole, fraction = ''] = amount.split('.');
  const paddedFraction = fraction.padEnd(decimals, '0').slice(0, decimals);
  return BigInt(whole + paddedFraction).toString();
}

async function payWithToken(
  tokenSymbol: 'USDT' | 'USDC' | 'USD1',
  amount: string,
  recipient: string
) {
  const owner = new Wallet(process.env.PRIVATE_KEY!);
  const token = TOKENS[tokenSymbol];
  const amountWei = toWei(amount, token.decimals);

  // Get smart wallet
  const walletRes = await fetch(
    `${FACILITATOR}/wallet/${owner.address}/master`
  );
  const { walletAddress } = await walletRes.json();

  // Verify
  const verifyRes = await fetch(`${FACILITATOR}/wallet/verify`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      walletAddress,
      transactions: [{ to: recipient, amount: amountWei, token: token.address }],
      paymentRequirements: { network: 'bsc' },
    }),
  });
  const { userOp, userOpHash } = await verifyRes.json();

  // Sign and settle
  const signature = await owner.signMessage(
    Buffer.from(userOpHash.slice(2), 'hex')
  );

  const settleRes = await fetch(`${FACILITATOR}/wallet/settle`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      userOp: { ...userOp, signature },
      signature,
    }),
  });

  const result = await settleRes.json();
  console.log(`Paid ${amount} ${tokenSymbol} via TX: ${result.txHash}`);
  return result;
}

// Usage
await payWithToken('USDT', '0.01', '0x...seller');
await payWithToken('USDC', '0.50', '0x...seller');
await payWithToken('USD1', '1.00', '0x...seller');
```

**Expected output:**
```
Paid 0.01 USDT via TX: 0x1111...aaaa
Paid 0.50 USDC via TX: 0x2222...bbbb
Paid 1.00 USD1 via TX: 0x3333...cccc
```

## 5. Balance Check Before Payment

Verify the smart wallet has sufficient funds before initiating a payment. Prevents failed transactions and wasted gas.

```ts
import { Wallet, Contract, JsonRpcProvider } from 'ethers';

const RPC = 'https://bsc-dataseed1.binance.org';
const FACILITATOR = 'https://facilitator.b402.ai';
const USDT = '0x55d398326f99059fF775485246999027B3197955';

async function payWithBalanceCheck(
  recipient: string,
  amountWei: string
) {
  const provider = new JsonRpcProvider(RPC);
  const owner = new Wallet(process.env.PRIVATE_KEY!, provider);

  // Get smart wallet address
  const walletRes = await fetch(
    `${FACILITATOR}/wallet/${owner.address}/master`
  );
  const { walletAddress, deploymentCompleted } = await walletRes.json();

  if (!deploymentCompleted) {
    throw new Error(
      `Wallet ${walletAddress} is not deployed. Deploy via /deploy/wallet/verify first.`
    );
  }

  // Check token balance on-chain
  const token = new Contract(
    USDT,
    ['function balanceOf(address) view returns (uint256)'],
    provider
  );

  const balance = await token.balanceOf(walletAddress);
  const required = BigInt(amountWei);

  console.log('Wallet:', walletAddress);
  console.log('Balance:', balance.toString(), 'wei');
  console.log('Required:', amountWei, 'wei');

  // Add 1% buffer for the facilitator fee
  const requiredWithBuffer = required + required / 100n;

  if (balance < requiredWithBuffer) {
    const deficit = requiredWithBuffer - balance;
    throw new Error(
      `Insufficient balance. Need ${requiredWithBuffer.toString()} wei ` +
      `(incl. ~1% fee buffer), have ${balance.toString()} wei. ` +
      `Deficit: ${deficit.toString()} wei.`
    );
  }

  console.log('Balance sufficient. Proceeding with payment...');

  // Verify
  const verifyRes = await fetch(`${FACILITATOR}/wallet/verify`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      walletAddress,
      transactions: [{ to: recipient, amount: amountWei, token: USDT }],
      paymentRequirements: { network: 'bsc' },
    }),
  });

  const verifyData = await verifyRes.json();

  if (!verifyData.isValid) {
    throw new Error(`Verification failed: ${JSON.stringify(verifyData)}`);
  }

  // Sign
  const signature = await owner.signMessage(
    Buffer.from(verifyData.userOpHash.slice(2), 'hex')
  );

  // Settle
  const settleRes = await fetch(`${FACILITATOR}/wallet/settle`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      userOp: { ...verifyData.userOp, signature },
      signature,
    }),
  });

  const result = await settleRes.json();

  if (result.success) {
    console.log('Payment successful. TX:', result.txHash);
  } else {
    throw new Error(`Settlement failed: ${JSON.stringify(result)}`);
  }

  return result;
}

// Usage
payWithBalanceCheck(
  '0x...seller_address',
  '10000000000000000' // 0.01 USDT
).catch(console.error);
```

**Expected output (success):**
```
Wallet: 0x...smart_wallet_address
Balance: 50000000000000000 wei
Required: 10000000000000000 wei
Balance sufficient. Proceeding with payment...
Payment successful. TX: 0xabcd...1234
```

**Expected output (insufficient balance):**
```
Wallet: 0x...smart_wallet_address
Balance: 5000000000000000 wei
Required: 10000000000000000 wei
Error: Insufficient balance. Need 10100000000000000 wei (incl. ~1% fee buffer),
have 5000000000000000 wei. Deficit: 5100000000000000 wei.
```
